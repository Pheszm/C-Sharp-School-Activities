create Database Ite_activity;

use Ite_activity;

create table Department_tbl(
Dep_Id INT PRIMARY KEY AUTO_INCREMENT,
Department varchar(100)
);

create table Brgy_tbl(
Brgy_Id INT PRIMARY KEY AUTO_INCREMENT,
Brgy varchar(100),
Municipality varchar(100),
Region varchar(100)
);

create table Profile(
Id INT PRIMARY KEY AUTO_INCREMENT,
LastName varchar(100),
FirstName varchar(100),
Middle_Initial varchar(2),
BirthDate varchar(20),
Dep_Id INT,
Brgy_Id INT,
FOREIGN KEY (Dep_Id) REFERENCES Department_tbl(Dep_Id),
FOREIGN KEY (Brgy_Id) REFERENCES Brgy_tbl(Brgy_Id)
);
